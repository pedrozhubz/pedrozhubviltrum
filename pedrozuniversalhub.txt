local Toggle = HomeTab:CreateToggle({
   Name = "Infinite jump",
   CurrentValue = false,
   Flag = "Togglejump", -- A flag is the identifier for the configuration file, make sure every element has a different flag if you're using configuration saving to ensure no overlaps
   Callback = function(Value)
   if Value then
	_G.JumpCon = game:GetService("UserInputService").JumpRequest:Connect(function()
		local h = game.Players.LocalPlayer.Character and game.Players.LocalPlayer.Character:FindFirstChildOfClass("Humanoid")
		if h then h:ChangeState(Enum.HumanoidStateType.Jumping) end
	end)
else
	if _G.JumpCon then
		_G.JumpCon:Disconnect()
		_G.JumpCon = nil
	end
end
   end,
})
