local OpTab = Window:CreateTab("OP", 4483362458) -- Title, Image

local Label = OpTab:CreateLabel("Note: You can't do most of things if any auto is turned on.", 4483362458, Color3.fromRGB(0, 0, 0), false) -- Title, Icon, Color, IgnoreTheme

local farming = false

local Toggle = OpTab:CreateToggle({
   Name = "Infinite XP and lots of money",
   CurrentValue = false,
   Flag = "XPmoney",

   Callback = function(Value)
      farming = Value
   end,
})

task.spawn(function()
   while true do
      if farming then
         local args = {
             [1] = "EnduranceTesting"
         }

         game:GetService("ReplicatedStorage")
             .Remotes
             .ClaimIncremental
             :FireServer(unpack(args))
      end

      task.wait()
   end
end)
